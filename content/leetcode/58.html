Solution

{{% code java %}}class Solution { public int lengthOfLastWord(String s) { int len = 0; for (int i = s.length() - 1; i >= 0; i--) { if (s.charAt(i) == ' ') { // account for trailing spaces if (len != 0) break; } else { len++; } } return len; } }{{% /code %}}
