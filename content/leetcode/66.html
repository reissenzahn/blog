Solution

{{% code java %}}class Solution { public int[] plusOne(int[] digits) { for (int i = digits.length - 1; i >= 0; i--) { if (digits[i] < 9) { digits[i]++; return digits; } digits[i] = 0; } // if we get here then we have 00..00 in the array and need to add a leading 1 int[] newDigits = new int[digits.length + 1]; newDigits[0] = 1; return newDigits; } }{{% /code %}}
