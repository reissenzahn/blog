--- title: "203. Remove Linked List Elements" date: 2021-02-22 draft: false ---

Solution

{{% code java %}}public class Solution { public ListNode removeElements(ListNode head, int val) { ListNode dummy = new ListNode(-1); dummy.next = head; ListNode curr = head; ListNode prev = dummy; while (curr != null) { if (curr.val == val) { prev.next = curr.next; } else { prev = prev.next; } curr = curr.next; } return dummy.next; } }{{% /code %}}
