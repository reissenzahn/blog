Solution

{{% code java %}}/** * public class TreeNode { * int val; * TreeNode left; * TreeNode right; * TreeNode() {} * TreeNode(int val) { this.val = val; } * TreeNode(int val, TreeNode left, TreeNode right) { * this.val = val; * this.left = left; * this.right = right; * } * } */ class Solution { public int maxDepth(TreeNode root) { if (root == null) return 0; return Math.max(this.maxDepth(root.right), this.maxDepth(root.left)) + 1; } }{{% /code %}}
