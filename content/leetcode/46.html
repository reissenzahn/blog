Solution

{{% code java %}}class Solution { public List> permute(int[] nums) { List> result = new ArrayList<>(); backtrack(nums, result, new ArrayList<>(), new boolean[nums.length]); return result; } private void backtrack(int[] nums, List> result, List perm, boolean[] used) { if (perm.size() == nums.length) { result.add(new ArrayList<>(perm)); return; } for (int i = 0; i < nums.length; i++) { if (used[i]) { continue; } perm.add(nums[i]); used[i] = true; backtrack(nums, result, perm, used); perm.remove(perm.size() - 1); used[i] = false; } } }{{% /code %}}
