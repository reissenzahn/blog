Solution

{{% code java %}}public int[] twoSum(int[] nums, int target) { // 1. two pass solution Map map = new HashMap<>(); for (int i = 0; i < nums.length; i++) { map.put(nums[i], i); } for (int i = 0; i < nums.length; i++) { int need = target - nums[i]; if (map.containsKey(need) && map.get(need) != i) { return new int[] {i, map.get(need)}; } } throw new IllegalArgumentException("No solution!"); // 2. one pass solution Map map = new HashMap(); for (int i = 0; i < nums.length; i++) { int need = target - nums[i]; if (map.containsKey(need)) { return new int[] {i, map.get(need)}; } map.put(nums[i], i); } throw new IllegalArgumentException("No solution!"); }{{% /code %}}
