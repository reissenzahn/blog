Solution

{{% code java %}}public ListNode reverseList(ListNode head) { // start at the head ListNode curr = head; ListNode prev = null; // continue until the end of the list while (curr != null) { // store the next node ListNode next = curr.next; // reverse the current reference curr.next = prev; // move to the next node prev = curr; curr = next; } return prev; }{{% /code %}}
