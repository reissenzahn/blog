Solution

{{% code java %}}class Solution { public String simplifyPath(String path) { Deque stack = new LinkedList<>(); for (String s : path.split("/")) { if (s.equals("..")) { if (!stack.isEmpty()) { stack.pop(); } } else if (!s.equals("") && !s.equals(".")) { stack.push(s); } } if (stack.isEmpty()) { return "/"; } StringBuilder sb = new StringBuilder(); while (!stack.isEmpty()) { sb.append("/"); sb.append(stack.pollLast()); } return sb.toString(); } }{{% /code %}}
