Solution

{{% code java %}}class Solution { public List findDisappearedNumbers(int[] nums) { // we want to an array to indicate if the value at that index is in nums but to save space we store the information back in nums using negative numbers // e.g. [4, 3, 2, 7, 8, 2, 3, 1] yields [-, -, -, -, +, +, -, -] for (int i = 0; i < nums.length; i++) { int val = Math.abs(nums[i]); if (nums[val - 1] > 0) { nums[val - 1] *= -1; } } List list = new ArrayList<>(); for (int i = 0; i < nums.length; i++) { if (nums[i] > 0) { list.add(i + 1); } } return list; } }{{% /code %}}
