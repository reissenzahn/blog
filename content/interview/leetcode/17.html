Solution

{{% code java %}}class Solution { private static final char[][] map = { null, null, {'a', 'b', 'c'}, {'d', 'e', 'f'}, {'g', 'h', 'i'}, {'j', 'k', 'l'}, {'m', 'n', 'o'}, {'p', 'q', 'r', 's'}, {'t', 'u', 'v'}, {'w', 'x', 'y', 'z'} }; public List letterCombinations(String digits) { LinkedList queue = new LinkedList(); if (digits.isEmpty()) { return queue; } queue.add(""); // go through the entire queue once for each digit for (int i = 0; i < digits.length(); i++) { int index = digits.charAt(i) - '0'; // go through the entire queue while (queue.peek().length() == i) { String s = queue.remove(); // add each possible character for (char c : map[index]) { queue.addLast(s + c); } } } return queue; } }{{% /code %}}
