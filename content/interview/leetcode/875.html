Solution

{{% code java %}}class Solution { public int minEatingSpeed(int[] piles, int h) { int lo = 1; int hi = this.maxPile(piles); while (lo < hi) { int mid = (lo + hi) / 2; if (this.canEatAll(piles, h, mid)) { hi = mid; } else { lo = mid + 1; } } return lo; } private boolean canEatAll(int[] piles, int h, int k) { int time = 0; for (int i = 0; i < piles.length; i++) { time += piles[i] / k; if (piles[i] % k != 0) time++; } return time <= h; } private int maxPile(int[] piles) { int max = piles[0]; for (int i = 0; i < piles.length; i++) { max = Math.max(max, piles[i]); } return max; } }{{% /code %}}
