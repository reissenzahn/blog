Promises


new Promise((resolve, reject) => {
  setTimeout(() => {
    if (Math.random() > 0.5) {
      resolve('Success!');
    } else {
      reject(new Error('Fail!'));
    }
  }, 1000);
})
.then((result) => console.log(result))
.catch((error) => console.error(error.message))
.finally(() => console.log('Done!'));
